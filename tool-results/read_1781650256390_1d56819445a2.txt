   133→// --- Where matching --------------------------------------------------------
   134→
   135→function matchValue(recordVal: any, condition: any): boolean {
   136→  if (condition === null || condition === undefined) {
   137→    return recordVal === null || recordVal === undefined
   138→  }
   139→  if (typeof condition === 'object' && !Array.isArray(condition)) {
   140→    if (condition.contains !== undefined) {
   141→      return String(recordVal ?? '').toLowerCase().includes(String(condition.contains).toLowerCase())
   142→    }
   143→    if (condition.gt !== undefined) return recordVal > condition.gt
   144→    if (condition.gte !== undefined) return recordVal >= condition.gte
   145→    if (condition.lt !== undefined) return recordVal < condition.lt
   146→    if (condition.lte !== undefined) return recordVal <= condition.lte
   147→    if (condition.in !== undefined) return Array.isArray(condition.in) && condition.in.includes(recordVal)
   148→    if (condition.not !== undefined) return !matchValue(recordVal, condition.not)
   149→  }
   150→  return recordVal === condition
   151→}
   152→
   153→function matchWhere(record: AnyRecord, where: AnyRecord | undefined): boolean {
   154→  if (!where || Object.keys(where).length === 0) return true
   155→  for (const [key, condition] of Object.entries(where)) {
   156→    if (key === 'AND') {
   157→      if (!Array.isArray(condition)) return false
   158→      if (!condition.every((sub: AnyRecord) => matchWhere(record, sub))) return false
   159→      continue
   160→    }
   161→    if (key === 'OR') {
   162→      if (!Array.isArray(condition)) return false
   163→      if (!condition.some((sub: AnyRecord) => matchWhere(record, sub))) return false
   164→      continue
   165→    }
   166→    if (key === 'NOT') {
   167→      if (typeof condition === 'object' && !Array.isArray(condition)) {
   168→        if (matchWhere(record, condition)) return false
   169→      }
   170→      continue
   171→    }
   172→    if (!matchValue(record[key], condition)) return false
   173→  }
   174→  return true
   175→}

   179→function applySelect(record: AnyRecord, select: AnyRecord | undefined): AnyRecord {
   180→  if (!select) return { ...record }
   181→  const out: AnyRecord = {}
   182→  for (const [key, include] of Object.entries(select)) {
   183→    if (key === '_count' && typeof include === 'object' && include) {
   184→      const selectCounts = (include as AnyRecord).select || {}
   185→      const counts: AnyRecord = {}
   186→      for (const [relKey] of Object.entries(selectCounts)) {
   187→        counts[relKey] = countRelation(record, relKey)
   188→      }
   189→      out._count = counts
   190→      continue
   191→    }
   192→    out[key] = record[key]
   193→  }
   194→  return out
   195→}

   197→function countRelation(record: AnyRecord, relKey: string): number {
   198→  // Map plural relation name back to model
   199→  const map: Record<string, string> = {
   200→    leads: 'lead',
   201→    contacts: 'contactHistory',
   202→    campaigns: 'prospectionCampaign',
   203→    notifications: 'notification',
   204→  }
   205→  const model = map[relKey] || relKey
   206→  const all = store[model] || []
   207→  if (relKey === 'leads' || relKey === 'campaigns' || relKey === 'notifications') {
   208→    return all.filter(r => r.userId === record.id).length
   209→  }
   210→  if (relKey === 'contacts') {
   211→    return all.filter(r => r.leadId === record.id).length
   212→  }
   213→  return all.length
   214→}

   216→function applyInclude(record: AnyRecord, include: AnyRecord | undefined): AnyRecord {
   217→  if (!include) return { ...record }
   218→  const out: AnyRecord = { ...record }
   219→  for (const [key, value] of Object.entries(include)) {
   220→    if (key === '_count' && typeof value === 'object' && value) {
   221→      const selectCounts = (value as AnyRecord).select || {}
   222→      const counts: AnyRecord = {}
   223→      for (const [relKey] of Object.entries(selectCounts)) {
   224→        counts[relKey] = countRelation(record, relKey)
   225→      }
   226→      out._count = counts
   227→      continue
   228→    }
   229→    if (key === 'user') {
   230→      const u = store.user.find(x => x.id === record.userId)
   231→      if (u && typeof value === 'object' && value !== null) {
   232→        out.user = applySelect(u, (value as AnyRecord).select)
   233→      } else {
   234→        out.user = u ? { ...u } : null
   235→      }
   236→      continue
   237→    }
   238→    if (key === 'lead') {
   239→      const l = store.lead.find(x => x.id === record.leadId)
   240→      if (l && typeof value === 'object' && value !== null) {
   241→        const nested = (value as AnyRecord).include || (value as AnyRecord).select
   242→        if ((value as AnyRecord).include) out.lead = applyInclude(l, (value as AnyRecord).include)
   243→        else if ((value as AnyRecord).select) out.lead = applySelect(l, (value as AnyRecord).select)
   244→        else out.lead = { ...l }
   245→      } else {
   246→        out.lead = l ? { ...l } : null
   247→      }
   248→      continue
   249→    }
   250→    if (key === 'preferences') {
   251→      out.preferences = store.preference.find(x => x.userId === record.id) || null
   252→      continue
   253→    }
   254→    if (key === 'creditTransactions') {
   255→      out.creditTransactions = store.creditTransaction
   256→        .filter(x => x.userId === record.id)
   257→        .sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime())
   258→      continue
   259→    }
   260→  }
   261→  return out
   262→}

   264→// --- Sorting ---------------------------------------------------------------
   265→
   266→function sortRecords(records: AnyRecord[], orderBy?: AnyRecord): AnyRecord[] {
   267→  if (!orderBy) return records
   268→  const entries = Object.entries(orderBy)
   269→  if (entries.length === 0) return records
   270→  return [...records].sort((a, b) => {
   271→    for (const [key, dir] of entries) {
   272→      const av = a[key]
   273→      const bv = b[key]
   274→      if (av === bv) continue
   275→      let cmp = 0
   276→      if (av === null || av === undefined) cmp = -1
   277→      else if (bv === null || bv === undefined) cmp = 1
   278→      else if (av < bv) cmp = -1
   279→      else if (av > bv) cmp = 1
   280→      if (cmp !== 0) return dir === 'desc' ? -cmp : cmp
   281→    }
   282→    return 0
   283→  })
   284→}

   286→// --- Model delegate --------------------------------------------------------
   287→
   288→class ModelDelegate {
   289→  constructor(private model: string) {}
   290→
   291→  private collection(): AnyRecord[] {
   292→    return store[this.model] || (store[this.model] = [])
   293→  }
   294→
   295→  async findMany(args?: AnyRecord): Promise<AnyRecord[]> {
   296→    let records = this.collection().filter(r => matchWhere(r, args?.where))
   297→    records = sortRecords(records, args?.orderBy)
   298→    if (args?.skip !== undefined) records = records.slice(args.skip)
   299→    if (args?.take !== undefined) {
   300→      records = args.take >= 0 ? records.slice(0, args.take) : records.slice(args.take)
   301→    }
   302→    return records.map(r => {
   303→      let out = r
